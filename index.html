Algorithm for File Updates in Python

Developed a Python algorithm that parses a file containing allowed IP addresses and removes ones that no longer have access to restricted content.

# Define a function to update the allowed IP list
def update_file(import_file, remove_list):
    with open(import_file, "r") as file:
        ip_addresses = file.read().split()

    for element in ip_addresses[:]:
        if element in remove_list:
            ip_addresses.remove(element)

    with open(import_file, "w") as file:
        file.write(" ".join(ip_addresses))

update_file("allow_list.txt", [
    "192.168.25.60",
    "192.168.140.81",
